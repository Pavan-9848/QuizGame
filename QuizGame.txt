import java.util.Scanner;

public class QuizGame {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int score = 0;

        System.out.println("----- Welcome to the Quiz Game -----");

        // Question 1

            score++;
        }
